Pattern

Pattern. Parameters : image, repeat (true, false, 'x', 'y'; true by default).

var pat = ctx.pattern('./img.png');

var pat2 = ctx.pattern('#image', 'x');

var i = new Image;
i.src = '../img.jpg';
var pat3 = ctx.pattern({
  image: i,
  repeat: false
});

Inline pattern

ctx.rect(0, 0, 1, 1, './img.jpg');
// link must begin with a '.' or 'http://'

ctx.rect(1, 1, 1, 1, {
  image: './png.jpg.gif',
  repeat: 'y'
});

repeat ( value : boolean / string )

Returns / changes. Values : true, false, 'x', 'y'.
